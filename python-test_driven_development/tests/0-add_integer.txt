import unittest

class TestAddInteger(unittest.TestCase):
    def test_add_integer_with_integers(self):
        result = add_integer(2, 3)
        self.assertEqual(result, 5)

    def test_add_integer_with_float(self):
        result = add_integer(2.5, 3.5)
        self.assertEqual(result, 6)

    def test_add_integer_with_default_value(self):
        result = add_integer(2)
        self.assertEqual(result, 100)

    def test_add_integer_with_non_integer(self):
        with self.assertRaises(TypeError):
            add_integer('a', 'b')

if __name__ == '__main__':
    unittest.main()
